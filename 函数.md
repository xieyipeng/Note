#include<stdio.h>

	int a[5];
	for (int i = 0; i < 5; i++) {
		a[i] = getchar();//ABCDE
	}
	for (int i = 0; i < 5; i++) {
		printf("%d\n", a[i]);//65 66 67 68 69
	}
